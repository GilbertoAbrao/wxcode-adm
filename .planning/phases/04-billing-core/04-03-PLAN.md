    1. **billing/webhook_router.py**: Create the webhook ingestion endpoint. This is a SEPARATE file from router.py because it has fundamentally different auth requirements (no JWT, uses Stripe-Signature header).

       ```python
       from __future__ import annotations

       import logging
       from typing import Annotated

       import stripe
       from fastapi import APIRouter, Depends, Header, HTTPException, Request

       from wxcode_adm.config import settings
       from wxcode_adm.tasks.worker import get_arq_pool

       logger = logging.getLogger(__name__)

       webhook_router = APIRouter(tags=["Webhooks"])


       async def get_raw_body(request: Request) -> bytes:
           """
           Read raw request body as bytes.
           CRITICAL: Must NOT parse as JSON — Stripe signature verification
           requires the exact wire bytes. Any JSON parsing produces different
           byte representation and breaks verification.
           """
           return await request.body()


       @webhook_router.post("/webhooks/stripe", status_code=200)
       async def stripe_webhook(
           stripe_signature: Annotated[str, Header(alias="stripe-signature")],
           body: bytes = Depends(get_raw_body),
       ) -> dict:
           """
           Stripe webhook ingestion endpoint.

           Fast path: verify signature, enqueue to arq, return 200.
           Processing happens asynchronously in the arq worker.
           Stripe retries on non-2xx, so returning 200 quickly is critical.

           Auth: Stripe-Signature header (NOT JWT). This endpoint must NOT
           use require_verified or any JWT auth dependency.
           """
           try:
               event = stripe.Webhook.construct_event(
                   payload=body,
                   sig_header=stripe_signature,
                   secret=settings.STRIPE_WEBHOOK_SECRET.get_secret_value(),
               )
           except ValueError:
               raise HTTPException(status_code=400, detail="Invalid payload")
           except stripe.SignatureVerificationError:
               raise HTTPException(status_code=400, detail="Invalid signature")

           # Enqueue with _job_id = Stripe event ID for arq-level deduplication.
           # arq atomic Redis transaction guarantees no duplicate enqueue while
           # the job is still queued or running.
           pool = await get_arq_pool()
           try:
               await pool.enqueue_job(
                   "process_stripe_event",
                   event["id"],
                   event["type"],
                   event["data"]["object"],
                   _job_id=event["id"],
               )
           finally:
               await pool.aclose()

           logger.info(f"Webhook received and enqueued: {event['type']} ({event['id']})")
           return {"received": True}
       ```

       **`async def process_stripe_event(ctx: dict, event_id: str, event_type: str, data_object: dict)`**:
       This is an arq job function — receives `ctx` dict with `session_maker` (set in worker startup hook).

       ```python
       async def process_stripe_event(ctx: dict, event_id: str, event_type: str, data_object: dict):
           """arq job: process one Stripe webhook event. Idempotent via DB check."""
           session_maker = ctx["session_maker"]

           async with session_maker() as db:
               try:
                   # DB-level idempotency check (outlasts arq result TTL)
                   existing = await db.execute(
                       select(WebhookEvent).where(WebhookEvent.stripe_event_id == event_id)
                   )
                   if existing.scalar_one_or_none():
                       logger.info(f"Webhook {event_id} already processed, skipping")
                       return

                   # Route to handler
                   if event_type == "checkout.session.completed":
                       await _handle_checkout_completed(db, data_object)
                   elif event_type == "customer.subscription.updated":
                       await _handle_subscription_updated(db, data_object)
                   elif event_type == "customer.subscription.deleted":
                       await _handle_subscription_deleted(db, data_object)
                   elif event_type == "invoice.paid":
                       await _handle_invoice_paid(db, data_object)
                   elif event_type == "invoice.payment_failed":
                       await _handle_payment_failed(db, data_object)
                   else:
                       logger.info(f"Unhandled webhook event type: {event_type}")

                   # Record processed event
                   db.add(WebhookEvent(
                       stripe_event_id=event_id,
                       event_type=event_type,
                       processed_at=datetime.now(timezone.utc),
                   ))
                   await db.commit()

               except Exception:
                   await db.rollback()
                   logger.exception(f"Failed to process webhook {event_id} ({event_type})")
                   raise  # Let arq retry
       ```

       **Event handlers** (private functions in billing/service.py):

       **`async def _handle_checkout_completed(db, data_object)`**:
       - Extract `customer` (stripe_customer_id) and `subscription` (stripe_subscription_id) from data_object.
       - Extract `tenant_id` and `plan_id` from `data_object["metadata"]`.
       - Query TenantSubscription by tenant_id.
       - Update: stripe_subscription_id, plan_id (UUID from metadata), status=SubscriptionStatus.ACTIVE.
       - Log: "Checkout completed for tenant {tenant_id}, plan {plan_id}".

       **`async def _handle_subscription_updated(db, data_object)`**:
       - Extract stripe_subscription_id from data_object["id"].
       - Query TenantSubscription by stripe_subscription_id.
       - If not found, log warning and return (may be a subscription we don't manage).
       - Map Stripe status to SubscriptionStatus: "active"/"trialing" -> ACTIVE, "past_due" -> PAST_DUE, "canceled"/"incomplete_expired" -> CANCELED. Default: log and keep current status.
       - Update current_period_start and current_period_end from data_object["current_period_start"] and ["current_period_end"] — these are Unix timestamps, convert with `datetime.fromtimestamp(ts, tz=timezone.utc)`.
       - If period changed (new period start differs from old), reset tokens_used_this_period to 0.

       **`async def _handle_subscription_deleted(db, data_object)`**:
       - Extract stripe_subscription_id from data_object["id"].
       - Query TenantSubscription by stripe_subscription_id. If not found, log warning and return.
       - Set status=SubscriptionStatus.CANCELED.
       - Log: "Subscription deleted for tenant {subscription.tenant_id}".

       **`async def _handle_invoice_paid(db, data_object)`**:
       - Extract stripe_subscription_id from data_object.get("subscription"). If None (one-off invoice), return.
       - Query TenantSubscription by stripe_subscription_id. If not found, return.
       - If status is PAST_DUE: restore to ACTIVE. Log: "Subscription restored for tenant {subscription.tenant_id}".
       - (Per CONTEXT: automatic restoration on invoice.paid, no manual intervention.)

       **`async def _handle_payment_failed(db, data_object)`**:
       - Extract stripe_subscription_id from data_object.get("subscription"). If None, return.
       - Query TenantSubscription by stripe_subscription_id. If not found, return.
       - Set status=SubscriptionStatus.PAST_DUE.
       - **Revoke JWT tokens** for all tenant members (per CONTEXT locked decision: "JWT tokens are revoked on payment failure"):
         Import Redis: `from wxcode_adm.common.redis_client import redis_client`.
         Import RefreshToken, TenantMembership from respective models.
         a. Query all TenantMembership.user_id WHERE tenant_id = subscription.tenant_id.
         b. Query all RefreshToken WHERE user_id IN (user_ids).
         c. For each token: set Redis key `auth:blacklist:jti:{token.token}` with TTL = ACCESS_TOKEN_TTL_HOURS * 3600 (from settings).
         d. Delete all RefreshToken rows for these users.
         e. Log: "Revoked tokens for {len(user_ids)} members of tenant {subscription.tenant_id}".
       - **Enqueue payment failure email** (notify owner + admins with billing_access):
         Query TenantMembership + User WHERE tenant_id AND (role=OWNER OR billing_access=True).
         For each: enqueue `send_payment_failed_email` arq job with email, tenant_name.

       ```python
       """arq job for sending payment failure notification emails."""
       import logging
       from fastapi_mail import MessageSchema, MessageType
       from wxcode_adm.auth.email import _get_mail_connection

       logger = logging.getLogger(__name__)

       async def send_payment_failed_email(ctx: dict, email: str, tenant_name: str):
           """
           Send payment failure notification email to tenant owner/admins.
           Follows the same pattern as auth/email.py send_verification_email.
           """
           subject = f"[WXCODE] Payment failed for {tenant_name}"
           body = (
               f"Payment has failed for your workspace '{tenant_name}'.\n\n"
               "Your workspace access to the wxcode engine has been restricted.\n"
               "Please update your payment method in the billing settings to restore access.\n\n"
               "You can still access wxcode-adm to manage your account and billing.\n\n"
               "— WXCODE Team"
           )
           logger.info(f"Payment failed email for {email} (tenant: {tenant_name})")

           try:
               fm = _get_mail_connection()
               message = MessageSchema(
                   subject=subject,
                   recipients=[email],
                   body=body,
                   subtype=MessageType.plain,
               )
               await fm.send_message(message)
           except Exception:
               logger.warning(f"Failed to send payment failure email to {email}", exc_info=True)
       ```
