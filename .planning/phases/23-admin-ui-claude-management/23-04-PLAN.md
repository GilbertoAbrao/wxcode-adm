    1. Add a localStorage key constant at the top:
       ```typescript
       const ADMIN_REFRESH_KEY = "wxk_admin_refresh";
       const ADMIN_EMAIL_KEY = "wxk_admin_email";
       ```

    2. Modify `setAdminTokens(access, refresh)` to ALSO persist the refresh token to localStorage:
       ```typescript
       export function setAdminTokens(access: string, refresh: string): void {
         _adminAccessToken = access;
         _adminRefreshToken = refresh;
         if (typeof window !== "undefined") {
           localStorage.setItem(ADMIN_REFRESH_KEY, refresh);
         }
       }
       ```
       NOTE: Only the refresh token is persisted — the access token is short-lived and will be obtained via refresh. This is the standard security pattern: refresh token in localStorage, access token in memory only.

    3. Modify `clearAdminTokens()` to ALSO clear localStorage:
       ```typescript
       export function clearAdminTokens(): void {
         _adminAccessToken = null;
         _adminRefreshToken = null;
         if (typeof window !== "undefined") {
           localStorage.removeItem(ADMIN_REFRESH_KEY);
           localStorage.removeItem(ADMIN_EMAIL_KEY);
         }
       }
       ```

    4. Add a new exported function to persist admin email:
       ```typescript
       export function setAdminEmail(email: string): void {
         if (typeof window !== "undefined") {
           localStorage.setItem(ADMIN_EMAIL_KEY, email);
         }
       }

       export function getStoredAdminEmail(): string | null {
         if (typeof window !== "undefined") {
           return localStorage.getItem(ADMIN_EMAIL_KEY);
         }
         return null;
       }
       ```

    5. Modify `refreshAdminTokens()` to also try loading from localStorage if _adminRefreshToken is null:
       At the top of the function, before the early return:
       ```typescript
       export async function refreshAdminTokens(): Promise<boolean> {
         let currentRefreshToken = _adminRefreshToken;
         if (!currentRefreshToken && typeof window !== "undefined") {
           currentRefreshToken = localStorage.getItem(ADMIN_REFRESH_KEY);
         }
         if (!currentRefreshToken) {
           return false;
         }
         // ... rest of function unchanged
       ```
       (Change `const` to `let` for currentRefreshToken)

    In frontend/src/providers/admin-auth-provider.tsx:

    1. Import the new functions:
       ```typescript
       import {
         isAdminAuthenticated,
         setAdminTokens,
         clearAdminTokens,
         getAdminRefreshToken,
         refreshAdminTokens,
         setAdminEmail as persistAdminEmail,
         getStoredAdminEmail,
       } from "@/lib/admin-auth";
       ```

    2. Replace the mount useEffect (the one that just does `setIsLoading(false)`) with session restore logic:
       ```typescript
       useEffect(() => {
         let cancelled = false;

         async function restoreSession() {
           // If already authenticated in memory, nothing to do
           if (isAdminAuthenticated()) {
             const storedEmail = getStoredAdminEmail();
             if (storedEmail) setAdminEmail(storedEmail);
             setIsLoading(false);
             return;
           }

           // Try to restore from localStorage via refresh
           const restored = await refreshAdminTokens();
           if (!cancelled) {
             if (restored) {
               const storedEmail = getStoredAdminEmail();
               if (storedEmail) setAdminEmail(storedEmail);
             }
             setIsLoading(false);
           }
         }

         restoreSession();

         return () => { cancelled = true; };
       }, []);
       ```
